

# 安装

```js
npm install art-template --save
```

# 首先在浏览器中使用

## 导入文件

```html
<script src='node_modules/art-template.lib/template-web.js'></script>
<script type='text/template' id='tel'>
	我的名字叫{{name}}
	我今年{{age}}岁了
	我喜欢{{ each hobbies }} {{ $value }} {{ /each }}
</script>
<script>
	var ret = template('tpl',{
    name:'jack',
    age:18,
    hobbies:[
      '唱歌',
      '看电影',
      '听歌'
    ]
  })
  log(ret)
</script>
```

```
{{ each hobbies }} {{ $value }} {{ /each }}
```

以上是循环语法

<font color='red'>模版引擎不关心内容</font>

# 在Node中使用(服务端渲染页面)

安装好后，使用require来导入

```js
let template = require('art-template')
```

使用render函数

```js
let tplStr = `	我的名字叫{{name}}
	我今年{{age}}岁了
	我喜欢{{ each hobbies }} {{ $value }} {{ /each }}`

var ret = template.render(tplStr,{
   name:'jack',
    age:18,
    hobbies:[
      '唱歌',
      '看电影',
      '听歌'
    ]
})
log(ret)
```

### 文件目录demo

```js
//httpdir.js
let http = require('http')
let serve = http.createServer()
let dir = 'E:/learn/js/node'
let template = require('art-template')
let fs = require('fs')
let ret
serve.on('request', function (req, res){
  fs.readFile('./template.html', function (err, data){
    if(err){
      return console.log('找不到文件');
      
    }
    fs.readdir(dir, function(err, files){
      if(err){
        return console.log('路径失效');
        
      }
      ret=template.render(data.toString(),{
        files
      })
    })
    res.end(ret)
  })
})
serve.listen('3000', function (){
  console.log('服务器开启....');
  
})
```

```html

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>
<body>
  {{each files}}
  <p>文件名：{{$value}}</p>
  {{/each}}
</body>
</html>
```
