/**
 * @param {string} paragraph
 * @param {string[]} banned
 * @return {string}
 */
var mostCommonWord = function(paragraph, banned) {
    let set = new Set(banned)
    let arr = paragraph.toLowerCase().match(/[a-z]+/g).filter(t => !set.has(t))
    let obj = {}
    let res = ''
    let max = 0
    for(let a of arr){
        obj[a] = (obj[a] || 0) + 1
        if (obj[a] > max) {
            res = a
            max = obj[a]
        }
    }
    return res
}
