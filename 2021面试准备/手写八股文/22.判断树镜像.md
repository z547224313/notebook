var isSymmetric = function(root) {
    function dfs(a,b) {
        if(a === null && b === null) return true
        if(a === null || b === null) return false
        else {
            return (a.val === b.val) && dfs(a.left,b.right) && dfs(a.right, b.left)
        }
    }
    return dfs(root,root)
};
