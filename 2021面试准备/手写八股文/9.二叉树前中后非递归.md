# 前

```js
var inorderTraversal = function(root) {
  let arr = []
  let res = []
  if(root) {
    arr.push(root)
  } else {
    return null
  }
  while(arr.length !== 0) {
    let temp = arr.pop()
    res.push(temp.val)
    if(temp.right) {
      arr.push(temp.right)
    }
    if(temp.left) {
      arr.push(temp.left)
    }
  }
  return res
};
```

var inorderTraversal = function(root) {
  let arr = []
  let res = []
  if(root) {
    arr.push(root)
  } else {
    return null
  }
  while(arr.length !== 0) {
    let temp = arr.pop()
    res.push(temp.val)
    if(temp.right) {
      arr.push(temp.right)
    }
    if(temp.left) {
      arr.push(temp.left)
    }
  }
  return res.reverse()
};
