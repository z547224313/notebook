function shen(obj){
  if(typeof obj!=="object"|| obj ===null || obj.constructor === RegExp || obj.constructor ===Date){
    return obj
  }
  let res = Array.isArray(obj)?[]:{}
  for(let key in obj){
    if(obj.hasOwnProperty(key)){
      res[key] = shen(obj[key])
    }
  }
  return res
}
